class Temperature
{ 
    private $title ;
    private $price ;
    private $flag ;

    public function __construct($title, $price, $flag )
    {
        $this->title = $title;
        $this->price = $price;
        $this->flag = $flag;
    }

    public function get_title()
    {
        return $this->title;
    } 
    public function get_price()
    {
        return $this->price;
    } 
    public function get_flag()
    {
        return $this->flag;
    } 
    public function average()
    {
        $total = 0;
        for ($i=0;$i<3;$i++) {
            $total += $this->_title->readTemp();
        }
        return $total/3;
    }
}
